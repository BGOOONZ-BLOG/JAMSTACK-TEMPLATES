  getInitialState() {
    return {
      disabled: true
    }
  },
  toggle(){
    this.setState({
      disabled: !this.state.disabled
    });
  },
  render() {
    return <div>
      <Switch disabled={this.state.disabled} />
      <br />
      <br />
      <button className="ant-btn ant-btn-primary" onClick={this.toggle}>Toggle disabled</button>
    </div>;
  }
});
