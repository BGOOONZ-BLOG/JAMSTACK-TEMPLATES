var SubMenu = Menu.SubMenu;

var Sider = React.createClass({
  getInitialState() {
    return {
      current: '1'
    }

    console.log('click ', e);
    this.setState({
      current: e.key
    });
  },
  render() {
    return <Menu onClick={this.handleClick}

                 selectedKeys={[this.state.current]}

        <Menu.Item key="1">选项1</Menu.Item>
