  children.push(<Option key={i.toString(36) + i}>{i.toString(36) + i}</Option>);
}

}

ReactDOM.render(
  <Select multiple
    style={{ width: 400 }}
    defaultValue={['a10', 'c12']} onChange={handleChange}>
